
grammar TestGrammar

abstract Operator(Word word)

Basic(Word value)
: .value

Complex(Keyword? value)
: ('hi' -> value)
: [.value]

List(Word[] words)
: (@Word -> words) (@Word -> words) (delim(@Word, ',') -> words)

Delimited(Word[] words)
: .words('.' '.')

Optional(Word? word)
: 'not_optional' .word
: 'optional' [.word]

Fail(Operator operator)
: .operator

TestGrammar(Stuff stuff)
: .stuff @EOF

union Stuff
: Basic
: Complex
: List
